					import Vue from 'vue';

					const Counter = Vue.component('vue-counter', {
					  props: { initial: Number },
					  data: function() {
					    return {
					      count: this.initial
					    }
					  },
					  template: `
					    <div class="centered">
					      <button @click="--count">
					        -
					      </button>
					      <span>{{count}}</span>
					      <button @click="++count">
					        +
					      </button>
					    </div>
					  `
					});

					new Vue({
					  el: '#vue-app',
					  components: { Counter },
					  template: '<vue-counter :initial="10" />'
					})
